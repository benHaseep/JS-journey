

Sort: 
```js
let names = [ "John", "Moaz", "Ali" ];

let nums = [8,3,5,6,2,5,9,1,4];

let mix = [8,3,5,"John","Ali" ,6,1,4 ]

  

console.log(names.sort()); // [ "Ali", "John", "Moaz" ]

console.log(nums.sort()); // [ 1, 2, 3, 4, 5, 5, 6, 8, 9 ]

console.log(mix.sort()); // [ 1, 3, 4, 5, 6, 8, "Ali", "John" ]
```

reverse:
```js
let names = [ "John", "Moaz", "Ali" ];

let nums = [8,3,5,6,2,5,9,1,4];

  

console.log(names.reverse()); // [ "Ali", "Moaz", "John" ]

console.log(nums.reverse()); // [ 4, 1, 9, 5, 2, 6, 5, 3, 8 ]
```
